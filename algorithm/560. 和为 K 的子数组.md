# 560. 和为 K 的子数组
``` js
/**
 * @param {number[]} nums
 * @param {number} k
 * @return {number}
 */
var subarraySum = function(nums, k) {
    let map = new Map()
    let preSum = 0
    let total = 0
    map.set(0,1)
    for(let i=0;i<nums.length;i++){
        preSum+=nums[i]
        let a = preSum-k
        if(map.get(a)){
            total+=map.get(a)
        }
        map.set(preSum,(map.get(preSum)||0)+1)
    }
    return total
};
```
