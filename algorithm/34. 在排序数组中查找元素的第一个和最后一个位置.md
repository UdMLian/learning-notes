# 34. 在排序数组中查找元素的第一个和最后一个位置

```js
/**
 * @param {number[]} nums
 * @param {number} target
 * @return {number[]}
 */
var searchRange = function(nums, target) {
    let right = nums.length-1
    let left = 0
    let c = -1
    function bsearch(left,right,isLeft){
        while(left<=right){
            let mid = Math.floor((left+right)/2)
            if(nums[mid]<target){
                left=mid+1
            }else if(nums[mid]>target){
                right=mid-1
            }else{
                c=mid
                if(isLeft){
                    right=mid-1
                }else{
                    left=mid+1
                }
            }
        }
        return c
    }
    let first = bsearch(left,right,true)
    if(first===-1) return [-1,-1]
    let last = bsearch(left,right,false)

    return [first,last]
};
```
