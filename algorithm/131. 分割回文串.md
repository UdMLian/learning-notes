# 131. 分割回文串

```js
/**
 * @param {string} s
 * @return {string[][]}
 */
var partition = function(s) {
    const isHuiwen = function(str,left,right){
        while(left<right){
            if(str[left]!==str[right]) return false
            left++
            right--
        }
        return true
    }

    const result = []

    const dfs=(start,track)=>{
        if(start===s.length) result.push([...track])

        for(let i = start;i<s.length;i++){
            if(!isHuiwen(s,start,i)) {
                continue
            }
            track.push(s.slice(start,i+1))
            dfs(i+1,track)
            track.pop()
        }
    }

    dfs(0,[])
    return result
};
```
