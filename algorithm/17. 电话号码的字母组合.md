# 17. 电话号码的字母组合

```js
/**
 * @param {string} digits
 * @return {string[]}
 */
var letterCombinations = function(digits) {
    if(digits.lengtg===0) return[]
    let map = {
        2:'abc',
        3:'def',
        4:'ghi',
        5:'jkl',
        6:'mno',
        7:'pqrs',
        8:'tuv',
        9:'wxyz'
    }
    let result = []
    let first = digits[0]
    let str = map[first]
    function dfs(index,track){
        if(digits.length===track.length){
            result.push(track.join(''))
            return
        }
        let next = digits[index]
        str=map[next]
        for(let char of str){
            track.push(char)
            dfs(index+1,track)
            track.pop()
        }
    }
    dfs(0,[])
    return result
};
```
