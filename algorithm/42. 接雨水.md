# 42. 接雨水
``` js
/**
 * @param {number[]} height
 * @return {number}
 */
var trap = function(height) {
    let stack = []
    let sum = 0
    for(let i=0;i<height.length;i++){
        let high = height[i]
        while(stack.length>0&&height[stack[stack.length-1]]<high){
            let bottom = height[stack.pop()]
            if (stack.length === 0) break;
            let left = stack[stack.length-1]
            let h = Math.min(height[left],high)-bottom
            sum+=h*(i-left-1)
        }
        stack.push(i)
    }
    return sum
};
```
