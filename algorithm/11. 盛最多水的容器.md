# 11. 盛最多水的容器
``` js
/**
 * @param {number[]} height
 * @return {number}
 */
var maxArea = function(height) {
    let max = 0
    let left = 0
    let right = height.length-1
    while(left<right){
        let heighter = Math.min(height[left],height[right])
        let current = (right-left)*heighter
        max=Math.max(current,max)
        if(height[left]<height[right]){
            left++
        }else{
            right--
        }
    }
    return max
};
```
