# 22. 括号生成

```js
/**
 * @param {number} n
 * @return {string[]}
 */
var generateParenthesis = function(n) {
    let result = []
    function dfs(left,right,track){
        if(track.length===n*2){
            result.push(track)
        }

        if(left<n) dfs(left+1,right,track+'(')
        if(right<left) dfs(left,right+1,track+')')
        
    }

    dfs(0,0,[])
    return result
};
```
