# 128. 最长连续序列
```js
/**
 * @param {number[]} nums
 * @return {number}
 */
var longestConsecutive = function(nums) {
    let set = new Set(nums)
    let max = 0
    for(const num of set){
        if(!set.has(num-1)){
            let currentNum = num
            let currentMax = 1
            while(set.has(currentNum+1)){
                currentMax++
                currentNum++
            }
            max=max<currentMax?currentMax:max
        }
    }
    return max
};
```
