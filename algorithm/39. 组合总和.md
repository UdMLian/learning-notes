# 39. 组合总和

```js
/**
 * @param {number[]} candidates
 * @param {number} target
 * @return {number[][]}
 */
var combinationSum = function(candidates, target) {
    let result = []
    function dfs(start,track,sum) {
        if(sum===target){
            result.push([...track])
            return
        }
        if(sum>target){
            return
        }
        for(let i=start;i<candidates.length;i++){
            track.push(candidates[i])
            dfs(i,track,sum+candidates[i])
            track.pop()
        }

    }
    dfs(0,[],0)
    return result
};
```
