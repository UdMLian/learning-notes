# 78. 子集

```js
/**
 * @param {number[]} nums
 * @return {number[][]}
 */
var subsets = function(nums) {
    let result=[]
    let track = []

    function dfs(start,track){
        result.push([...track])

        for(let i=start;i<nums.length;i++){
            track.push(nums[i])
            dfs(i+1,track)
            track.pop()
        }
    }
    dfs(0,track)
    return result
};
```
